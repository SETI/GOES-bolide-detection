#------------------------------------------------------------------------------
outputDir = ./output/G16

#------------------------------------------------------------------------------
# detectionType : str
#   What are we detecting in the GLM data?
#   (e.g. 'bolides', 'gigantic_jets')

#------------------------------------------------------------------------------

#------------------------------------------------------------------------------

#
# Base name is extended with '.txt'
#------------------------------------------------------------------------------
processingHistoryBaseName = processing_history

# processing history.
# WARNING: THIS WILL DELETE ALL FILES CURRENTLY LOCATED AT <outputDir>
#------------------------------------------------------------------------------
forceReprocess = True

#------------------------------------------------------------------------------
# multiProcessEnabled - If true, then use GNU parallel to distribute the processing
#------------------------------------------------------------------------------

#------------------------------------------------------------------------------

#------------------------------------------------------------------------------
n_cores_aggregation = 20

#------------------------------------------------------------------------------

#------------------------------------------------------------------------------
deleteDailyDirs = False
deleteDailyNetCdfFiles = True

#------------------------------------------------------------------------------
# deleteDailyDatabases - If True, then delete all the daily database files.

#------------------------------------------------------------------------------
deleteDailyDatabases = True

#------------------------------------------------------------------------------
# doNotProcessCurrentDay - If True then do not process any data from the current day.

#------------------------------------------------------------------------------
doNotProcessCurrentDay = True

#------------------------------------------------------------------------------
# delayNDaysForL0Data - (int) Do not process the daily data if the Level 0
# data is not present. But only check up to this number of days back in time. This
# means for all days older than this many N days old, process the data
# irrespective of if the Level 0 data is available
#------------------------------------------------------------------------------
delayNDaysForL0Data = 10

# data before this data. Format: yyyy-mm-dd
#------------------------------------------------------------------------------
#startDate = 2018-06-21
startDate = ''

#------------------------------------------------------------------------------
# endDate - Gives the stoping date for processing the data. Do not process any 
# data after this data. Format: yyyy-mm-dd
#------------------------------------------------------------------------------
#endDate = 2018-07-23
endDate = ''
